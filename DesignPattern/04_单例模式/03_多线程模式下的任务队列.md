```cpp
#include<iostream>
#include<queue>
#include<mutex>
#include<thread>

using namespace std;

class TaskQueue
{
public:
    // 禁用拷贝构造与赋值运算符，确保实例唯一
    TaskQueue(const TaskQueue& t) = delete;
    TaskQueue& operator = (const TaskQueue& t) = delete;

    // 静态成员函数：外部获取唯一实例的官方接口
    static TaskQueue* getInstance()
    {
        return m_taskQ;
    }

    void print()
    {
        cout << "我是单例对象的一个成员函数..." << endl;
    }

    //定义一个函数判断任务队列是否为空
    bool isEmpty()
    {
        lock_guard<mutex> locker(m_mutex);
        bool flag = m_data.empty();
        return flag;
    }
    //添加任务
    void addTask(int node)
    {
        lock_guard<mutex> locker(m_mutex);
        m_data.push(node);
    }
    //删除任务
    bool popTask()
    {
        lock_guard<mutex> locker(m_mutex);
        //删除队头元素
        if(m.data.empty())
        {
            return false;
        }
        m_data.pop();
        return true;
    }
    //取任务（不删除任务）
    int takeTask()
    {
        lock_guard<mutex> locker(m_mutex);
        if(m_data.empty())
        {
            return -1;
        }
        int data = m_data.front();
        return data;
    }

private:
    // 私有化构造函数，防止外部 new
    TaskQueue() = default;
    // 静态指针成员变量
    static TaskQueue* m_taskQ;
    //定义任务队列
    queue<int> m_data;
    mutex m_mutex;
};

// 在类外部进行初始化（程序启动时执行）
TaskQueue* TaskQueue::m_taskQ = new TaskQueue;

int main()
{
    TaskQueue* taskQ = TaskQueue::getInstance();
    //生产者
    thread t1([=]()
    {
        for(int i = 0; i < 100; i++)
        {
            taskQ->addTask(i + 100);
            cout << "+++ push data: " <<i + 100 << ", threadID:" << this_thread::get_id() << endl;
            this_thread::sleep_for(chrono::milliseconds(500));
        }
    })
    //消费者
    thread t2([=]()
    {
        this_thread::sleep_for(chrono::milliseconds  (100));
        while(!taskQ->isEmpty())
        {
            int num = taskQ->takeTask();
            cout << "+++ push data: " << num << ", threadID:" << this_thread::get_id() << endl;
            taskQ->popTask();
            this_thread::sleep_for(chrono::milliseconds(1000));
        }
    })
    t1.join();
    t2.join();
    return 0;
}
```
