```cpp
#include<iostream>
using namespace std;

//产品类的父类
class AbstractSmile
{
public:
    virtual void tranform() = 0;
    virtual void ability() = 0;
    virtual ~AbstractSmile() {}
};

class sheepSmile : public AbstractSmile
{
public:
    void tranform() override
    {
        cout << "变成人兽 -- 山羊人形态..." << endl;
    }
    void ability() override
    {
        cout << "将手臂变成绵阳角的招式 -- 巨羊角" << endl;
    }
};

class lionSmile : public AbstractSmile
{
public:
    void tranform() override
    {
        cout << "变成人兽 -- 狮子人形态..." << endl;
    }
    void ability() override
    {
        cout << "可以喷火" << endl;
    }
};

class batSmile : public AbstractSmile
{
public:
    void tranform() override
    {
        cout << "变成人兽 -- 蝙蝠人形态..." << endl;
    }
    void ability() override
    {
        cout << "万剑归宗能力" << endl;
    }
};

//定义工厂类 - 父类
class AbstractFactory
{
public:
    virtual AbstractSmile* createSmile() = 0;
    virtual ~AbstractFactory() {}
};

//生产山羊的恶魔果实工厂类
class sheepFactory : public AbstractFactory
{
public:
    AbstractSmile* createSmile()
    {
        return new sheepSmile;
    }
    ~sheepFactory() {
        cout << "sheepFactory被析构了..." << endl; 
    }
}

//生产狮子的恶魔果实的工厂类
class lionFactory : public AbstractFactory
{
public:
    AbstractSmile* createSmile()
    {
        return new lionSmile;
    }
    ~lionFactory() {
        cout << "lionFactory被析构了..." << endl; 
    }
}

//生产蝙蝠的恶魔果实的工厂类
class batFactory : public AbstractFactory
{
public:
    AbstractSmile* createSmile()
    {
        return new batSmile;
    }
    ~batFactory() {
        cout << "batFactory被析构了..." << endl; 
    }
}

int main()
{
    AbstractFactory *factory = new lionFactory;
    AbstractSmile* obj = factory->createSmile();
    obj->tranform();
    obj->ability();
    delete obj;
    delete factory;
    return 0;
}
```
